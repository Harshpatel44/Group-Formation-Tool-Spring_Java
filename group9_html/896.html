18     @Configuration  
19     @EnableWebSecurity
20     public class SecurityConfiguration extends WebSecurityConfigurerAdapter
21     {
22     	@Override
23        public void configure(WebSecurity web) throws Exception
24     	{
25        	web.ignoring().antMatchers("/resources/**");
26        }
27        
28     	@Override  
29     	public void configure(HttpSecurity http) throws Exception
30     	{
31     		http.authorizeRequests()
32     			.antMatchers("/public/**", "/**").permitAll()
33     			.antMatchers("/admin/**").hasRole("ADMIN")
34     			.anyRequest().authenticated()
35     			.and().formLogin()
36     				.loginPage("/login")
37     				.permitAll()
38     			.and().logout()
39     				.permitAll()
40     		;
41     	}
42     	
43     	@Override
44     	protected AuthenticationManager authenticationManager() throws Exception
