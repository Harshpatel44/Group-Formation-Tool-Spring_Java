19         @Override
20         public List<String> checkPasswordValidation(String password, Map<String, String> errors) {
21             List<String> policyErrors = new ArrayList<String>();
22             UserPasswordPolicy passwordPolicy = UserPasswordPolicy.getInstance();
23             UserPasswordPolicyStatus passwordPolicyStatus = UserPasswordPolicyStatus.getInstance();
24             if (passwordPolicyStatus.getMinLength() == 1) {
25                 if (password.length() < passwordPolicy.getMinLength()) {
26                     policyErrors.add("Password should have minimum " + +passwordPolicy.getMinLength() + " letters");
27                 }
28             }
29             if (passwordPolicyStatus.getMaxLength() == 1) {
30                 if (password.length() > passwordPolicy.getMaxLength()) {
31                     policyErrors.add("Password should have maximum " + passwordPolicy.getMaxLength() + " letters");
32                 }
33             }
34             if (passwordPolicyStatus.getMinUpperCaseLetter() == 1) {
35                 if (password.chars().filter((s) -> Character.isUpperCase(s)).count() < passwordPolicy
36                         .getMinUpperCaseLetter()) {
37                     policyErrors.add("Password should have minimum " + +passwordPolicy.getMinUpperCaseLetter()
38                             + " uppercaseLetters ");
39                 }
40             }
41             if (passwordPolicyStatus.getMinLowerCaseLetter() == 1) {
42                 if (password.chars().filter((s) -> Character.isLowerCase(s)).count() < passwordPolicy
43                         .getMinLowerCaseLetter()) {
44                     policyErrors.add("Password should have minimum " + passwordPolicy.getMinLowerCaseLetter()
45                             + " lowerrcaseLetters");
46                 }
47             }
48             if (passwordPolicyStatus.getMinNoOfSymbols() == 1) {
49                 if (password.split("[!_/-@#$%^&*()\\[\\]|;',./{}\\\\:\"<>?]", -1).length - 1 < passwordPolicy
50                         .getMinNoOfSymbols()) {
51                     policyErrors.add("Password should have minimum " + +passwordPolicy.getMinNoOfSymbols() + " symbols");
52                 }
53             }
54             if (passwordPolicyStatus.getNotAllowedCharacters() == 1) {
55                 if (StringUtils.containsAny(passwordPolicy.getNotAllowedCharacters(), password)) {
56                     policyErrors.add(
57                             "Following characters are not allowed in password " + passwordPolicy.getNotAllowedCharacters());
58                 }
59             }
60             if (policyErrors.size() > 0) {
61                 errors.put("confirmPassword", String.join(";;", policyErrors));
62             }
63             return policyErrors;
64         }
