63     	private Map<String, String> checkAllValidations(IUser user) {
64     		Map<String, String> errors = new HashMap<String, String>();
65     		if (user.getFirstName() == null || user.getFirstName().isEmpty()) {
66     			errors.put("firstName", "first name cannot be  null or empty");
67     		}
68     		if (user.getLastName() == null || user.getLastName().isEmpty()) {
69     			errors.put("lastName", "last name cannot be null or empty");
70     		}
71     		if (user.getBannerId() != null) {
72     			if (!user.getBannerId().isEmpty() && user.getBannerId().length() != 9) {
73     				errors.put("bannerId", "BannerId length is should be equal to 9");
74     			}
75     		} else {
76     			errors.put("bannerId", "BannerId cant be null or empty");
77     		}
78     		if (user.getEmailId() == null || user.getEmailId().isEmpty()) {
79     			errors.put("emailId", "Email cannot be  null or empty");
80     		} else if (!validateEmail(user.getEmailId())) {
81     			errors.put("emailId", "Enter valid Email");
82     		}
83     		if (user.getContactNumber() == null || user.getContactNumber().isEmpty()) {
84     			errors.put("contactNumber", "Contact Number cannot be  null or empty");
85     		} else if (user.getContactNumber().length() != 10) {
86     			errors.put("contactNumber", "Contact Number should have only 10 digits");
87     		}
88     		if (user.getPassword() == null || user.getPassword().isEmpty()) {
89     			errors.put("password", "Password cannot be null or empty");
90     		}
91     		if (user.getConfirmPassword() == null || user.getConfirmPassword().isEmpty()) {
92     			errors.put("confirmPassword", "Confirm Password cannot be  null or empty");
93     		}
94     		if (user.getPassword() != null) {
95     			if (!user.getPassword().equals(user.getConfirmPassword())) {
96     				errors.put("confirmPassword", "Passwords and confirm password Does not match");
97     			} else {
98     				Injector.instance().getUserPasswordPolicyService().checkPasswordValidation(user.getPassword(), errors);
99     			}
100     		}
101     		return errors;
102     	}
