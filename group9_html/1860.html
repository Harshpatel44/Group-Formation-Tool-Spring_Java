45             List<String> validationErrors = iUserPasswordPolicyService.checkPasswordValidation(newPassword, errors);

52             passNumber = service.getPasswordPolicyNumber();

54             for (String password : oldPasswords) {
55                 if (encryption.passwordMatch(newPassword, password)) {
56                     model.addAttribute("Error", "New password cannot be same as the old password");
57                     return "newPassword";
58                 }
59             }
60             update = service.updatePassword(bannerID, newPassword);
61             if (update == false) {
62                 model.addAttribute("Error", "Error updating the password");
63                 return "newPassword";
